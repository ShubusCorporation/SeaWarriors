pragma solidity 0.8.30;

import {ERC721} from "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import {ERC721Burnable} from "@openzeppelin/contracts/token/ERC721/extensions/ERC721Burnable.sol";
import {ERC721Pausable} from "@openzeppelin/contracts/token/ERC721/extensions/ERC721Pausable.sol";
import {ERC721URIStorage} from "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
import {Ownable} from "@openzeppelin/contracts/access/Ownable.sol";
import {ReentrancyGuard} from "@openzeppelin/contracts/utils/ReentrancyGuard.sol";
import "@openzeppelin/contracts/utils/Strings.sol";

contract SeaWarriors is ERC721, ERC721URIStorage, ERC721Pausable, Ownable, ERC721Burnable, ReentrancyGuard {
    using Strings for uint256;
    
    error InsufficientFunds();
    error WrongmetadataId();
    error DivisionByZero();
    error WithdrawalFailed();

    uint256 private _nextTokenId;
    uint256 private _totalPayments;
    uint256 private _totalSales;
    uint256 private constant TOTAL_PICTURES = 13;
    uint256 private constant MIN_PAYMENT = 0.0001 ether;

    // Events
    event Purchase(
        address indexed buyer,
        uint256 indexed tokenId,
        uint256 amount,
        uint256 metadataId
    );
    event Withdrawal(address indexed to, uint256 amount);

    constructor(address initialOwner)
        ERC721("Sea Warriors", "ABYSSWARS")
        Ownable(initialOwner)
    {}

    function _baseURI() internal pure override returns (string memory) {
        return "ipfs://Qmf2UPHvAqCXk2kMgPquSacNQWdckpHRGKRikevdbK6SG9/";
    }

    function pause() public onlyOwner {
        _pause();
    }

    function unpause() public onlyOwner {
        _unpause();
    }

    function safeMint(address to, uint256 metadataId)
        public
        onlyOwner
        returns (uint256)
    {
        if (metadataId == 0 || metadataId > TOTAL_PICTURES) {
            revert WrongmetadataId();
        }
        uint256 tokenId = _nextTokenId++;
        _safeMint(to, tokenId);
        _setTokenURI(tokenId, metadataId.toString());
        return tokenId;
    }

    function buy() 
        public
        payable
        nonReentrant
        whenNotPaused
    {
        if (msg.value < MIN_PAYMENT) {
            revert InsufficientFunds();
        }
        
        uint256 tokenId = _nextTokenId++;
        uint256 metadataId = getmetadataId(msg.value);
        
        // This check should never fail if algorithm is correct, but safety first
        if (metadataId == 0 || metadataId > TOTAL_PICTURES) {
            revert WrongmetadataId();
        }

        unchecked {
            _totalPayments += msg.value;
            _totalSales++;
        }
        
        _safeMint(msg.sender, tokenId);
        _setTokenURI(tokenId, metadataId.toString());
        
        emit Purchase(msg.sender, tokenId, msg.value, metadataId);
    }

    /**
     * @notice Withdraw contract balance to owner
     */
    function withdraw() public onlyOwner {
        uint256 balance = address(this).balance;
        if (balance == 0) {
            revert InsufficientFunds();
        }
        
        (bool success, ) = payable(owner()).call{value: balance}("");
        if (!success) {
            revert WithdrawalFailed();
        }
        
        emit Withdrawal(owner(), balance);
    }

    /**
     * @notice Optimized version - combines array creation and search in single loop
     * @dev Gas optimization: removed double loop and memory array storage
     */
    function getmetadataId(uint256 currentPayment) private view returns(uint256) {
        uint256 numerator = 90;
        uint256 denominator = 100;

        if (_totalSales > 0) {
            numerator = getNumerator(currentPayment, numerator);
        }

        uint256 initial = 1250;
        uint256 curr = initial;
        uint256 summ;
        
        // First pass: calculate sum without storing array
        for(uint256 i = 0; i < TOTAL_PICTURES; i++) {
            curr = curr * numerator / denominator;
            summ += curr;
        }

        // Generate random value in range [1, summ] to avoid edge case with 0
        uint256 random = getRandom(summ);
        if (random == 0) {
            random = 1; // Ensure we don't get 0
        }
        
        // Second pass: find the metadataId without storing array
        curr = initial;
        uint256 currentSumm;
        
        for(uint256 i = 0; i < TOTAL_PICTURES; i++) {
            curr = curr * numerator / denominator;
            currentSumm += curr;

            if (currentSumm >= random) {
                return i + 1;
            }
        }
        
        // Fallback (should never reach here if algorithm is correct)
        return 1;
    }

    /**
     * @notice Calculate numerator based on payment amount
     * @dev Fixed division by zero issue
     */
    function getNumerator(uint256 currentPayment, uint256 numerator) private view returns(uint256) {
        uint256 avg = _totalPayments / _totalSales;
        bool isBonus = currentPayment > avg;
        uint256 rem;

        if (isBonus) {
            rem = currentPayment - avg;
        } else {
            rem = avg - currentPayment;
        }
        
        uint256 avg5p = avg / 20;
        
        // FIX: Prevent division by zero
        if (avg5p == 0) {
            // If average is too small, use default numerator
            return numerator;
        }

        if (rem > avg5p) {
            uint256 coeff = rem * 5 / avg5p;

            if (isBonus) {
                numerator += coeff;
            } else {
                if (numerator > coeff) {
                    numerator -= coeff;
                } else {
                    numerator = 10;
                }
            }
        }
        return numerator;
    }

    /**
     * @notice Generate pseudo-random number
     * @dev WARNING: This is still predictable/manipulable. For production, use Chainlink VRF
     * @param N Maximum value for random number
     * @return Random number in range [1, N]
     */
    function getRandom(uint256 N) private view returns (uint256) {
        if (N == 0) {
            return 1;
        }
        
        // Improved randomness by adding more entropy sources
        uint256 randomHash = uint256(
            keccak256(
                abi.encodePacked(
                    block.timestamp,
                    block.number,
                    msg.sender,
                    tx.origin,
                    block.prevrandao,
                    gasleft() // Additional entropy
                )
            )
        );
        
        // Return value in range [1, N] instead of [0, N]
        return (randomHash % N) + 1;
    }

    // The following functions are overrides required by Solidity.

    function _setTokenURI(uint256 tokenId, string memory metadataId) 
        internal 
        override(ERC721URIStorage) 
    {
        metadataId = string.concat(metadataId, ".json");
        super._setTokenURI(tokenId, metadataId);
    }

    function _update(address to, uint256 tokenId, address auth)
        internal
        override(ERC721, ERC721Pausable)
        returns (address)
    {
        return super._update(to, tokenId, auth);
    }

    function tokenURI(uint256 tokenId)
        public
        view
        override(ERC721, ERC721URIStorage)
        returns (string memory)
    {
        return super.tokenURI(tokenId);
    }

    function supportsInterface(bytes4 interfaceId)
        public
        view
        override(ERC721, ERC721URIStorage)
        returns (bool)
    {
        return super.supportsInterface(interfaceId);
    }
}
